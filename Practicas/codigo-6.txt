interface IGuardar
{
	void Guardar(string g);
}

interface IMostrar
{
	void Mostrar(string m);
}

{

	public void Guardar(string g)
	{
		Console.WriteLine("Guardando: " + g);
	}

	public void Mostrar(string m)
	{
		Console.WriteLine("Mostrando: " + m);
	}
	
}

class Program
{
	static void Main()
	{
		Archivo a = new Archivo();

		IGuardar guardar = a;
		IMostrar mostrar = a;

	}
}
